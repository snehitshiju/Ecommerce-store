# Q5. Remove outliers using IQR and recalc IQR + boxplot
df_iqr = df.copy()
Q1 = df['Air Quality Index'].quantile(0.25)
Q3 = df['Air Quality Index'].quantile(0.75)
IQR = Q3 - Q1
lower, upper = Q1 - 1.5*IQR, Q3 + 1.5*IQR
df_iqr = df_iqr[(df_iqr['Air Quality Index'] >= lower) & (df_iqr['Air Quality Index'] <= upper)]
new_IQR = df_iqr['Air Quality Index'].quantile(0.75) - df_iqr['Air Quality Index'].quantile(0.25)
print("New IQR after removing outliers:", new_IQR)
plt.boxplot(df_iqr['Air Quality Index'])
plt.title("Boxplot - AQI (IQR Outliers Removed)")
plt.show()

# Q12. Plot CI for 25 Trials (n=500, CI=95%)
plt.figure(figsize=(10,6))
pop_mean = aqi.mean()
for i in range(25):
    s = aqi.sample(n=500, random_state=i)
    m, sd = s.mean(), s.std(ddof=1)
    se = sd / math.sqrt(500)
    ci = stats.norm.interval(0.95, loc=m, scale=se)
    plt.plot([ci[0], ci[1]], [i, i], 'b-', lw=2)
    plt.plot(m, i, 'ro')
plt.axvline(pop_mean, color='g', linestyle='--')
plt.title("95% CI for 25 Trials (n=500)")
plt.xlabel("Air Quality Index")
plt.ylabel("Trial")
plt.show()

z = (sample_mean - mu) / (sigma / math.sqrt(n))
print("Z-score:", z)
if z > z_crit:
    print("Reject Null Hypothesis  Claim supported (mean improved)\n")
else:
    print("Fail to Reject Null Hypothesis  Claim not supported\n")
